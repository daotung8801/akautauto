AKA_ASSERT_TRUE(value) Assert that value is TRUE (non-zero)
AKA_ASSERT_FALSE(value) Assert that value is FALSE (zero)
AKA_ASSERT_EQUAL(actual, expected) Assert that actual = = expected
AKA_ASSERT_NOT_EQUAL(actual, expected)) Assert that actual != expected
AKA_ASSERT_LOWER(actual, boundary) Assert that actual < boundary
AKA_ASSERT_LOWER_OR_EQUAL(actual, boundary) Assert that actual <= boundary
AKA_ASSERT_GREATER(actual, boundary) Assert that actual > boundary
AKA_ASSERT_GREATER_OR_EQUAL(actual, boundary) Assert that actual >= boundary
AKA_ASSERT_PTR_EQUAL(actual, expected) Assert that pointers actual = = expected
AKA_ASSERT_PTR_NOT_EQUAL(actual, expected) Assert that pointers actual != expected
AKA_ASSERT_PTR_NULL(value) Assert that pointer value == NULL
AKA_ASSERT_PTR_NOT_NULL(value) Assert that pointer value != NULL
AKA_ASSERT_STRING_EQUAL(actual, expected) Assert that strings actual and expected are equivalent
AKA_ASSERT_STRING_NOT_EQUAL(actual, expected) Assert that strings actual and expected differ
AKA_ASSERT_DOUBLE_EQUAL(actual, expected, granularity) Assert that |actual - expected| <= |granularity|
Math library must be linked in for this assertion.
AKA_ASSERT_DOUBLE_NOT_EQUAL(actual, expected, granularity) Assert that |actual - expected| > |granularity|
Math library must be linked in for this assertion.
